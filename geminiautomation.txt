### ✅ METHOD 1 — Click by text (ALWAYS try this first)
python -c "
import pyautogui, pytesseract

img = pyautogui.screenshot()
data = pytesseract.image_to_data(img, output_type=pytesseract.Output.DICT)
target = 'Search'  # <-- Replace with your target text
found = False
for i, text in enumerate(data['text']):
    if target.lower() in text.lower() and int(data['conf'][i]) > 50:
        x = data['left'][i] + data['width'][i] // 2
        y = data['top'][i] + data['height'][i] // 2
        print(f'Clicking [{text}] at ({x}, {y})')
        pyautogui.click(x, y)
        found = True
        break
if not found:
    print(f'NOT FOUND: {target}')
"

### ✅ METHOD 3 — Image template match (for icons)
python -c "
import pyautogui, cv2, numpy as np

img = pyautogui.screenshot()
img_np = cv2.cvtColor(np.array(img), cv2.COLOR_RGB2GRAY)
template = cv2.imread(r'C:\Users\Admin\Documents\For_automation\ref.png', 0)
res = cv2.matchTemplate(img_np, template, cv2.TM_CCOEFF_NORMED)
_, val, _, loc = cv2.minMaxLoc(res)
if val > 0.8:
    h, w = template.shape
    cx, cy = loc[0] + w//2, loc[1] + h//2
    print(f'Found at ({cx},{cy}) confidence={val:.2f}')
    pyautogui.click(cx, cy)
else:
    print(f'No match. Best confidence: {val:.2f}')
"
