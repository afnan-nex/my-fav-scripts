Run this ONCE before each decision. No file saving — use memory buffer:

import pyautogui, pytesseract, cv2, numpy as np
from PIL import Image
import io

# Capture to memory (NO file saving)
buf = io.BytesIO()
img = pyautogui.screenshot()
img.save(buf, format='PNG')
buf.seek(0)
print('Screenshot captured in memory')

# Get all visible text and their positions
data = pytesseract.image_to_data(img, output_type=pytesseract.Output.DICT)
for i, text in enumerate(data['text']):
if text.strip():
x, y, w, h = data['left'][i], data['top'][i], data['width'][i], data['height'][i]
print(f'TEXT: [{text}] at center=({x + w//2}, {y + h//2})')
"

## SMART CLICKING — PREFER THESE OVER RAW COORDINATES

### Method 1: Click by finding text on screen (BEST — most accurate)
python -c "
import pyautogui, pytesseract
from PIL import Image

img = pyautogui.screenshot()
data = pytesseract.image_to_data(img, output_type=pytesseract.Output.DICT)
target = 'OK' # <-- Change this to the button/text you want to click
for i, text in enumerate(data['text']):
if target.lower() in text.lower():
x = data['left'][i] + data['width'][i] // 2
y = data['top'][i] + data['height'][i] // 2
print(f'Found [{target}] at ({x}, {y}) — clicking')
pyautogui.click(x, y)
break
else:
print(f'Text [{target}] not found on screen')
"

### Method 2: Click by image template matching (for icons/buttons without text)
python -c "
import pyautogui, cv2, numpy as np
from PIL import Image

# Take screenshot and convert for OpenCV
img = pyautogui.screenshot()
img_np = np.array(img)
gray = cv2.cvtColor(img_np, cv2.COLOR_RGB2GRAY)

# Load your saved reference image (only time you use a file — pre-saved icon/button crop)
template = cv2.imread(r'C:\Users\Admin\Documents\For_automation\button_ref.png', 0)
result = cv2.matchTemplate(gray, template, cv2.TM_CCOEFF_NORMED)
_, max_val, _, max_loc = cv2.minMaxLoc(result)

if max_val > 0.8: # Confidence threshold
h, w = template.shape
cx, cy = max_loc[0] + w // 2, max_loc[1] + h // 2
print(f'Match found at ({cx}, {cy}) with confidence {max_val:.2f}')
pyautogui.click(cx, cy)
else:
print(f'No confident match found (best: {max_val:.2f})')
"
